local AutoFarmSection = Tabs.AutoFarm:AddSection("Thu Thập Tài Nguyên")

local AutoChopToggle = AutoFarmSection:AddToggle("AutoChop", {
	Title = "Auto Chặt Cây 🌲",
	Description = "Tự động chặt cây lấy gỗ",
	Default = false
})

local AutoCollectToggle = AutoFarmSection:AddToggle("AutoCollect", {
	Title = "Auto Nhặt Quả 🍓",
	Description = "Tự động nhặt quả mọng và nấm",
	Default = false
})

local AutoFishToggle = AutoFarmSection:AddToggle("AutoFish", {
	Title = "Auto Câu Cá 🎣",
	Description = "Tự động câu cá ở sông/hồ",
	Default = false
})

local AutoGardenToggle = AutoFarmSection:AddToggle("AutoGarden", {
	Title = "Auto Làm Vườn 🌻",
	Description = "Tự động trồng trọt và thu hoạch",
	Default = false
})

-- Biến toàn cục
_G.AutoChop = false
_G.AutoCollect = false
_G.AutoFish = false
_G.AutoGarden = false

AutoChopToggle:OnChanged(function(Value)
	_G.AutoChop = Value
	if Value then
		coroutine.wrap(function()
			while _G.AutoChop do
				task.wait(2)
				chopTrees()
			end
		end)()
	end
end)

AutoCollectToggle:OnChanged(function(Value)
	_G.AutoCollect = Value
	if Value then
		coroutine.wrap(function()
			while _G.AutoCollect do
				task.wait(3)
				collectResources()
			end
		end)()
	end
end)

AutoFishToggle:OnChanged(function(Value)
	_G.AutoFish = Value
	if Value then
		coroutine.wrap(function()
			while _G.AutoFish do
				task.wait(5)
				goFishing()
			end
		end)()
	end
end)

AutoGardenToggle:OnChanged(function(Value)
	_G.AutoGarden = Value
	if Value then
		coroutine.wrap(function()
			while _G.AutoGarden do
				task.wait(4)
				tendGarden()
			end
		end)()
	end
end)

-- Nút điều khiển

	Title = "Bật Tất Cả 🟢",
	Description = "Bật tất cả tính năng auto farm",

		_G.AutoChop = true
		_G.AutoCollect = true
		_G.AutoFish = true
		_G.AutoGarden = true

		AutoGardenToggle:SetValue(true)
	end
})

		_G.AutoGarden = false
		AutoChopToggle:SetValue(false)
		AutoCollectToggle:SetValue(false)
		AutoFishToggle:SetValue(false)
		AutoGardenToggle:SetValue(false)
	end
})

local NhanVatSection = Tabs.NhanVat:AddSection("Tùy Chỉnh Nhân Vật")

local WalkSpeedSlider = NhanVatSection:AddSlider("WalkSpeed", {
	Title = "Tốc Độ Di Chuyển 🏃",
	Description = "Chỉnh tốc độ chạy/bước",
	Default = 16,
	Min = 16,
	Max = 50,
	Rounding = 1,
	Callback = function(Value)
		local humanoid = game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
		if humanoid then
			humanoid.WalkSpeed = Value
		end
	end
})

local JumpPowerSlider = NhanVatSection:AddSlider("JumpPower", {
	Title = "Sức Mạnh Nhảy 🦘",
	Description = "Chỉnh độ cao khi nhảy",
	Default = 50,
	Min = 50,
	Max = 100,
	Rounding = 1,
	Callback = function(Value)
		local humanoid = game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
		if humanoid then
			humanoid.JumpPower = Value
		end
	end
})

NhanVatSection:AddToggle("NoClip", {
	Title = "Đi Xuyên Vật 🔮",
	Description = "Đi xuyên qua vật thể",
	Default = false,
	Callback = function(Value)
		_G.NoClip = Value
		if Value then
			enableNoClip()
		else
			disableNoClip()
		end
	end
})

NhanVatSection:AddToggle("InfJump", {
	Title = "Nhảy Vô Hạn 🌈",
	Description = "Nhảy liên tục không giới hạn",
	Default = false,
	Callback = function(Value)
		_G.InfiniteJump = Value
		if Value then
			game:GetService("UserInputService").JumpRequest:connect(function()
				if _G.InfiniteJump then
					game:GetService("Players").LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):ChangeState("Jumping")
				end
			end)
		end
	end
})
