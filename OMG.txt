-- Functions
local function CreateButton(text, position, callback)
    local Button = Instance.new("TextButton")
    Button.Name = text .. "Button"
    Button.Parent = ScrollFrame
    Button.BackgroundColor3 = Color3.fromRGB(35, 35, 55)
    Button.BorderSizePixel = 0
    Button.Position = position
    Button.Size = UDim2.new(0.94, 0, 0, 45)
    Button.Font = Enum.Font.GothamSemibold
    Button.Text = "üéÆ " .. text
    Button.TextColor3 = Color3.fromRGB(255, 255, 255)
    Button.TextSize = 14
    
    -- Hi·ªáu ·ª©ng hover OMG style
    Button.MouseEnter:Connect(function()
        game:GetService("TweenService"):Create(Button, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(45, 45, 70),
            TextColor3 = Color3.fromRGB(255, 255, 0)
        }):Play()
    end)
    
    Button.MouseLeave:Connect(function()
        game:GetService("TweenService"):Create(Button, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(35, 35, 55),
            TextColor3 = Color3.fromRGB(255, 255, 255)
        }):Play()
    end)
    
    Button.MouseButton1Click:Connect(callback)
    
    return Button
end

-- Cheat Functions cho 99 Dem trong Rung
local function GodMode()
    local character = Player.Character
    if character and character:FindFirstChild("Humanoid") then
        character.Humanoid.MaxHealth = math.huge
        character.Humanoid.Health = math.huge
    end
end

local function InfiniteStamina()
    while wait(1) do
        local character = Player.Character
        if character then
            -- Gi·∫£ s·ª≠ stamina ƒë∆∞·ª£c l∆∞u trong leaderstats ho·∫∑c playerstats
            local leaderstats = Player:FindFirstChild("leaderstats")
            if leaderstats then
                local stamina = leaderstats:FindFirstChild("Stamina")
                if stamina then
                    stamina.Value = 100
                end
            end
        end
    end
end

local function SpeedHack()
    local character = Player.Character
    if character and character:FindFirstChild("Humanoid") then
        character.Humanoid.WalkSpeed = 50
    end
end

local function NoClip()
    local character = Player.Character
    if character then
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end
end
